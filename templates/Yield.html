In [20]:
ML_Model = []
r2 = []
mse = []
rmse = []
mae = []

#function to call for storing the results
def storeResults(model, a,b,c,d):
    ML_Model.append(model)
    r2.append(round(a, 3))
    mse.append(round(b, 3))
    rmse.append(round(c, 3))
    mae.append(round(d, 3))
